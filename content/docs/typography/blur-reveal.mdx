## Usage

```tsx
// src/Root.tsx
import { Composition } from "remotion";
import { BlurReveal } from "@/components/remocn/blur-reveal";

const BlurRevealScene = () => (
  <BlurReveal text="Hello, world" blur={12} fontSize={72} />
);

export const RemotionRoot = () => (
  <Composition
    id="BlurReveal"
    component={BlurRevealScene}
    durationInFrames={90}
    fps={30}
    width={1280}
    height={720}
  />
);
```

### With Backdrop

The component renders transparent — supply the background via `Backdrop`. It ships single-theme; edit the copied file to re-theme colors.

Pair with `Backdrop` to place the text inside a full-frame fill with a rounded, shadowed content frame:

```tsx
import { Composition } from "remotion";
import { BlurReveal } from "@/components/remocn/blur-reveal";
import { Backdrop } from "@/components/remocn/backdrop";

const BlurRevealScene = () => (
  <Backdrop fill={{ type: "color", value: "#ffffff" }}>
    <BlurReveal text="Hello, world" blur={12} fontSize={72} />
  </Backdrop>
);

export const RemotionRoot = () => (
  <Composition
    id="BlurReveal"
    component={BlurRevealScene}
    durationInFrames={90}
    fps={30}
    width={1280}
    height={720}
  />
);
```
